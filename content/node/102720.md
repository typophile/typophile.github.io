  body: "Further:\r\n\r\n\" = quotedbl = 0022\r\n' = quotesingle = 0027\r\n\u201C
    = quotedblleft = 201C\r\n\u201D = quotedblright = 201D\r\n\u2018 = quoteleft =
    2018\r\n\u2019 = quoteright =2019"
