feature calt {
sub @minuscules' @capitales by @grandes; #first level of sub.

Try this:

feature calt {

lookup calt1 { sub @minuscules' @capitales by @grandes; #first level of sub. sub @capitales @minuscules' by @grandes; } calt1;

lookup calt2 { sub @minuscules' @grandes by @moyennes; #second level of sub. sub @grandes @minuscules' by @moyennes; } calt2;

lookup calt3 { sub @minuscules' @moyennes by @petites; sub @moyennes @minuscules' by @petites; } calt3;

} calt;
