I use so one script (saved as file in my /.FontForge/python/ directory) for tests. Script : save copy of my font as temp.sfd, remove overlaps, corect direction etc., generate font (without hints), copy it (or overwrite) to my local fonts directory (on Linux) and remove temp files. 
# -*- coding: utf-8 -*-
import shutil;
import os;

def genTestOTF(arg, ft):
   localFontdir='/.local/share/fonts/';
   fDir=os.path.expanduser("~")+localFontdir;
   ftActive=fontforge.activeFont();
   ftActive.save('temp.sfd');
   ft=fontforge.open('temp.sfd');
   aLayer=ft.layers[ft.activeLayer].name;
   ft.encoding='UnicodeBMP';
   ft.selection.all();
   for g in ft.selection.byGlyphs:
      g.removeOverlap();
      g.correctDirection();
      g.canonicalStart();
      g.canonicalContours();
   fontNazwa = ft.fontname;
   plik=fontNazwa+'_test.otf';
   ft.generate(plik, flags=('opentype','dummy-dsig','apple','no-hints','no-flex'),layer=aLayer)
   shutil.copy(plik,fDir+plik);
   ft.close();
   os.remove(plik);
   os.remove('temp.sfd');

fontforge.registerMenuItem(genTestOTF,None,None,"Font",None,"Generator","Quick test-otf");

another system, different FontForge version, most probably another Python version... :-) Ok, most simplest way to make it, and avoid saveAs: script bellow first save your work, then remove overlaps etc., generate .otf and revert your original font file (ff should ask you). 
# -*- coding: utf-8 -*-
import os;

def genTestOTF(arg, ft):
   ft.save();
   ft.encoding='UnicodeBMP';
   ft.selection.all();
   for g in ft.selection.byGlyphs:
      g.removeOverlap();
      g.correctDirection();
      g.canonicalStart();
      g.canonicalContours();
   plik=ft.fontname+'_test.otf';
   ft.generate(plik, flags=('opentype','dummy-dsig','apple','no-hints','no-flex'));
   ft.selection.none();
   ft.revert();

fontforge.registerMenuItem(genTestOTF,None,None,"Font",None,"Generator","Quick test-otf");
